#include <iostream>

using namespace std;

struct st_Client_Info
{
	string Account_Number;
	string Pin_Code;
	string Full_Name;
	string Phone_Number;
	string Account_Balance;

};

string Join_String(vector <string>& vNames, const string& delim)
{

	string S1 = "";

	for (string& s : vNames)
	{
		S1 = S1 + s + delim;
	}

	return S1.substr(0, S1.length() - delim.length());
}

string Convert_Recorde_to_Line(const st_Client_Info & Client_Info)
{

	vector <string> vClient_Info;

	vClient_Info.push_back(Client_Info.Account_Number);
	vClient_Info.push_back(Client_Info.Pin_Code);
	vClient_Info.push_back(Client_Info.Full_Name);
	vClient_Info.push_back(Client_Info.Phone_Number);
	vClient_Info.push_back(Client_Info.Account_Balance);


	return Join_String(vClient_Info, "#//#");
}
