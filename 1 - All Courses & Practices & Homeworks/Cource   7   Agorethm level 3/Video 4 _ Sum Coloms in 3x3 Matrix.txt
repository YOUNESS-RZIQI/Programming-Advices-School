int Randam_Number_In_Range(int From, int To)
{

	int RandNum = rand() % (To - From + 1) + From;

	return RandNum;
}

void Fill_Two_Dimensional_Arrays(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			Arr[i][j] = Randam_Number_In_Range(1, 100);
		}

	}


}

int Sum_Coloms_of_Matrix(int Arr[3][3], const short& Row_Number, const short& Coloms)
{

	int sum = 0;
	for (int j = 0; j < Row_Number; j++)
	{
		sum += Arr[j][Coloms];
	}

	return sum;
}

void Print_Colms_sum_of_Matrix(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Coloms; i++)
	{
		cout << "Sum of Colms [" << i + 1 << "] is : " << Sum_Coloms_of_Matrix(Arr, Rows, i) << "\n";
	}

}

void Print_Matrix(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			//printf("%d\t", Arr[i][j]);
			cout << setw(3) << Arr[i][j] << "    ";
		}

		cout << endl;

	}


}


int main()
{
	srand((unsigned)time(NULL));

	int arr[3][3];
	Fill_Two_Dimensional_Arrays(arr, 3, 3);
	cout << "The 3x3 array result : \n\n";
	Print_Matrix(arr, 3, 3);
	cout << "\n\n" << endl;
	Print_Colms_sum_of_Matrix(arr, 3, 3);

	cout << "\n\n" << endl;



	return 0;
}
