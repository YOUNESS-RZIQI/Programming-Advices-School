float Calculat_Circle_Area_By_Arbitrary_Triangle(float a, float b , float c)
{

	float Area,P , HI;
	const float PI = 3.141592653589793238;

	P = (a + b + c) / 2;

	HI = (a * b * c) / (4 * sqrt(P * (P - a) * (P - b) * (P - c))) ;

	Area = PI * pow(HI, 2);


	return Area;

}

{

	cout << "\n   The Area is : " << Result << endl << endl;

}
