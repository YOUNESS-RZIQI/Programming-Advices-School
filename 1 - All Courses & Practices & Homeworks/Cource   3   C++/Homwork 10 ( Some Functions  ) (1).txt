#include <iostream>
#include <cmath>
using namespace std;

int main()
{

	float a, b, c;
	
	const float PI = 3.14;

	cout << "Pleas enter the \"a\" side : ";
	cin >> a;
	cout << "Pleas enter the \"b\" side : ";
	cin >> b;
	cout << "Pleas enter the \"c\" side : ";
	cin >> c;

	const float p = (a + b + c) / 2;
	float calculaion = (a * b * c) /  (  4 * sqrt( p * (p - a) * (p - b) * (p - c))   )  ;
	float Area = PI * pow(calculaion, 2) ;

	cout << "The result is : " << round(Area) << endl;


	return 0;
}
