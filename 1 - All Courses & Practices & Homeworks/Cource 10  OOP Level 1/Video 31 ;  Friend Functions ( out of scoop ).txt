#include <iostream>
using namespace std;


class clsA
{

private:

	int Var1;

	void Hi1
	()
	{
		cout << "\nHi\n";
	}


protected:

	int Var2;

	void Hi2()
	{
		cout << "\nHi\n";
	}


public:

	clsA()
	{
		Var1 = 10;
		Var2 = 20;
		Var3 = 30;
	}

	int Var3;

	void Hi3()
	{
		cout << "\nHi\n";
	}

	friend int MySum(clsA A);
};
