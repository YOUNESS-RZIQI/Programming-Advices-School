/*
                       Title : Count Spesifc litter in a string .
*/


#include "clsString.h"



int Count_Spesifc_litter(string s,char letter)
{
    int counter = 0;

    for (int i = 0; i < s.length() ; i++)
    {
        if (s[i] == letter)
            counter++;
    }

    return (counter);
}

int main(void)
{

   
    cout << Count_Spesifc_litter("AA bbb", 'b');
    cout << endl;

    

    return 0;
}
