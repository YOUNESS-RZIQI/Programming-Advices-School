    Trim Prototype : 
        void Trim(string S , char seprator);

#include <iostream>
using namespace std;

void    Trim(string &S , char Seprator)
{
    int i = 0;
    int j = (S.length() - 1);

    while ( S[i] == Seprator && S[i])
            i++;

    if ( i == j + 1)
    {
        S = "";
        return ;
    }

    while (S[j] == Seprator && S[j])
            j--;

    S = S.substr(i, j - i + 1);

}
