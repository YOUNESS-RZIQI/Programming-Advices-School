#include "clsString.h"

static char Invert_letter_case(char c)
{
    return isupper(c) ? tolower(c) : toupper(c);
}



int main(void)
{


    cout << "result \'A\' : " << Invert_letter_case('A') << endl;
    cout << "result \'a\' : " << Invert_letter_case('a') << endl;

    return 0;
}
