    static bool IsNumBetween(int a, int from , int to)
    {
        if ( from > to)
            clsUtil::Swap(from , to);
        return (a >= from && a <= to);
    }

        if ( from > to)

        if ( clsDate::IsDate1BeforeDate2(To,from))
                clsDate::SwapDates(from,To);
        
        return (clsDate::IsDate1AfterDate2(date,from) && clsDate::IsDate1BeforeDate2(date,To) );
