static void Upper_first_leter(string &s)
{
    short is_first_letter = true;

    for (int i = 0; i < s.length(); i++)
    {
        if( s[i] != ' ' && is_first_letter)
        {
            if(islower(s[i]))
                s[i] = toupper(s[i]);
        }
        is_first_letter = (s[i] == ' '? true : false);
    }
}
