#include <iostream>
#include <vector>
using namespace std;

vector <string> Split(string s,string delemeter)
{
    vector <string> spleted;
    int d_pos = 0;
    string new_word;
  
    while ((d_pos = s.find(delemeter)) != std::string::npos)
    {
        new_word = s.substr(0,d_pos);
        if (new_word != "")
            spleted.push_back(new_word);

        s.erase(0, d_pos + delemeter.length());
    }
    if (s != "")
        spleted.push_back(s);
    
    return (spleted);
}

string ReverseWordsInString_My_Version(string S1)
{
    vector <string> S_Words = Split(S1, " ");
    string New_S;


    for (int i  = (S_Words.size() - 1); i >= 0  ; i--)
    {
        New_S += S_Words[i];
        if ( i > 0 )
        New_S += " ";

    }
    return (New_S);
}
