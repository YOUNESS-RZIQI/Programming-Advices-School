#include <iostream>
#include <cmath>
using namespace std;


float Read_Number()
{
	float Num = 0;

	cout << "Please enter The Number : ";
	cin >> Num;

	return Num;
}

float Get_Fraction(float Number)
{
	return Number - int(Number);
}

float My_ceil(float Number)
{
	if (abs(Get_Fraction(Number)) > 0)
		if (Number > 0)
			return int(Number) + 1;
		else
			return int(Number);
	else
		return Number;
}


int main()
{

	float Number = Read_Number();

	cout << "\nMy Ceil  : " << My_ceil(Number);
	cout << "\nC++ Ceil : " << ceil(Number) << endl;

	return 0;
}
