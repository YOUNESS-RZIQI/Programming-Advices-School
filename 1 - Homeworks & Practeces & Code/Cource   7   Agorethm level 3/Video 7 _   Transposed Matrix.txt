void Transposed_Matrix(int Arr[3][3],int Arr_Transposd[3][3], const short& Rows, const short& Coloms )
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			Arr_Transposd[j][i] = Arr[i][j];
		}

	}

}

void Print_Matrix(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			//printf("%d\t", Arr[i][j]);
			cout << setw(3) << Arr[i][j] << "    ";
		}

		cout << endl;

	}


}

void Fill_2_D_Arr_W_order_Numbers(int Arr[3][3], const short& Rows, const short& Coloms)
{

	short Count = 0;

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			Count++;
			Arr[i][j] = Count;
		}

	}


}
