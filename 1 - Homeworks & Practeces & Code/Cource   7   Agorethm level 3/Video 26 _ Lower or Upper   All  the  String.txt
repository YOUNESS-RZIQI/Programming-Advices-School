
#include <iostream>
#include <iomanip>
#include <vector>
#include <string>
#include <fstream>
using namespace std;

string Read_String(string Message)
{

	string Word = "";
	cout << Message << endl;
	getline(cin, Word);

	return Word;
}

string Apper_All_Word_Letters( string  Word)
{

	for (int i = 0; i < Word.length(); i++)
	{
		Word[i] = toupper(Word[i]);
	}

	return Word;
}

string Lower_All_Word_Letters(string  Word)
{

	for (int i = 0; i < Word.length(); i++)
	{
		Word[i] = tolower(Word[i]);
	}

	return Word;
}

int main()
{

	string Word = "";

	Word = Read_String("Please Enter Your  Word  : ");

	cout << "\n\nThe all Word Letters in Capital : \n";
	Word = Apper_All_Word_Letters(Word);
	cout << Word << endl;

	cout << "\n\nThe all Word Letters in Smale : \n";
	Word = Lower_All_Word_Letters(Word);
	cout << Word << endl;


	return 0;
}
