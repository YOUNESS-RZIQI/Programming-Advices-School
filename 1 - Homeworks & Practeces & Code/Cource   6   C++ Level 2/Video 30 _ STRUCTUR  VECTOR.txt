float Read_Number(string Message)
{
	float Number;
	cout << Message;
	cin >> Number;
	while (cin.fail())
	{
		// user didn't input a number
		cin.clear();
		cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
		cout << "Invalid Number, Enter a valid one :";
		cin >> Number;
	}
	return Number;
}
