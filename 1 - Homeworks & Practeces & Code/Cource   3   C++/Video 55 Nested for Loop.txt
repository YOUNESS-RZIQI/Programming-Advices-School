#include <iostream>
using namespace std;


void Maltuplation_Calculation()
{

	for (int i = 0; i <= 10; i++)
	{
		cout << " i = " << i << "\n\n---------------\n\n" ;


		for (int j = 1; j <= 10; j++)
		{

			cout  << i << "*" << j << " = " << i * j << endl << endl ;

		}

		cout << "---------------\n\n";

	}



}


int main()
{

	Maltuplation_Calculation();

	return 0;
}
