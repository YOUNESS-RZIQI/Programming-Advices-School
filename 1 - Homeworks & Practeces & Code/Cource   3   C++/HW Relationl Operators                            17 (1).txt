#include <iostream>
using namespace std;

int main()
{
	int A, B;

	
	cout << "Pleas enter   the Firs Number \'A\' : \n";
	cin >> A;

	cout << "Pleas enter the Second Number \'B\' : \n";
	cin >> B;

	cout << A << "  = " << B << "  is : " << (A == B) << endl;
	cout << A << " != " << B << "  is : " << (A != B) << endl;
	cout << A << "  > " << B << "  is : " << (A > B) << endl;
	cout << A << "  < " << B << "  is : " << (A < B) << endl;
	cout << A << " >= " << B << "  is : " << (A >= B) << endl;
	cout << A << " <= " << B << "  is : " << (A <= B) << endl;


	return 0;
}
