#include <iostream>

int Enter_Positive_Number(string Massege)
{

	int Number = 0;

	do
	{

		cout << Massege;
		cin >> Number;
		cout << endl;


	} while (Number <= 0);


	return Number;
}

void Print_Inverted_Number_Pattern(int Number)
{

	for (int j = Number; 1 <= j; j--)
	{

		for (int i = 1; j >= i; i++)
		{

			cout << j;

		}

		cout << endl;
		

	}

}


int main()
{

	Print_Inverted_Number_Pattern(Enter_Positive_Number("Please enter a positive Number : "));

	return 0;
