int Read_Number_In_Range(string Message, int From, int To)
{

	int Number = 0;

	do
	{
		cout << Message;
		cin >> Number;

	} while (Number > From && Number > To);

	return Number;

}

{

void Print_Array(int Arr[100], int ArrLength)
{


	for (int i = 0; i < ArrLength; i++)
	{

		cout << Arr[i] << " ";

	}

	cout << endl;


}

void Copy_Array(int Arr_Origin[100], int Arr_copy[100], int ArrLength)
{


	for (int i = 0; i < ArrLength; i++)
	{

		Arr_copy[i] = Arr_Origin[i];

	}



}

short Number_Position(int Arr[100], int ArrLength, int Num_to_Search)
{


	for (int i = 0; i < ArrLength; i++)
	{
		if (Arr[i] == Num_to_Search)
		{

			return i;

		}

	}

	return -1;

}

bool Is_Number_Exist_in_Array(int Arr[100], int ArrLength, int Num_to_Search)
{
		return Number_Position( Arr,  ArrLength,  Num_to_Search) != -1 ;
}
