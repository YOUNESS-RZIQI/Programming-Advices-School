using namespace std;

int NumberOfDaysInAMonth(short Year, short Month)
{
	if (Month <= 0 || Month > 12)
		return 0;

	if (Month == 2)
		return Check_Lib::is_leap_Year(Year) ? 29 : 28;

	/*else if (Month == 1 || Month == 3 || Month == 5 || Month == 7 || Month == 8 || Month == 10 || Month == 12)
		return 31;*/


	short arr31Days[] = { 1,3,5,7,8,10,12 };

	for (short i = 0; i < 7; i++)
	{
		if (arr31Days[i] == Month)
			return 31;
	}

	return 30;
}

short Days_from_begining_of_Year(const short& Year, const short& Month, const short& Day)
{

	short Counter_of_Days = 0;

	for (short i = 1; i < Month; i++)
	{
		Counter_of_Days += NumberOfDaysInAMonth( i , Year);
	}
	return Counter_of_Days + Day;
}
int main()
{
	
	short Day = Input_Lib::Read_Positive_Number  ("Day   : ");
	short Month = Input_Lib::Read_Positive_Number("Month : ");
	short Year = Input_Lib::Read_Positive_Number ("Year  : ");

	cout << "Days from begining of Year is : " << Days_from_begining_of_Year(Year, Month, Day);
	

	system("pause>0");
	return 0;
}
