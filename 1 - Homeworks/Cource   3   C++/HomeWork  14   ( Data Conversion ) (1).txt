#include <iostream>
#include <string>
using namespace std;

int main()
{

	string str = "123.456";

	//conversion string to integer
	int Num_int = stoi(str);

	//conversion string to double
	double Num_double = stod(str);

	//conversion strin to float 
	float Num_float = stof(str);


	cout << Num_int << endl;
	cout << Num_double << endl;
	cout << Num_float << endl;




	return 0;
}
