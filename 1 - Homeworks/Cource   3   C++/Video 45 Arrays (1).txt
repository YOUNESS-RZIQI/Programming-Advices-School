#include <iostream>
#include <iomanip>
using namespace std;




int main()
{

	float Grade[3] ;

	cout << "Pleas enter your 1 Grade ?\n";
	cin >> Grade[0];
	cout << "Pleas enter your 2 Grade ?\n";
	cin >> Grade[1];
	cout << "Pleas enter your 3 Grade ?\n";
	cin >> Grade[2];

	

	float Average = (Grade[0] + Grade[1] + Grade[2]) / 3;

	cout << "********************************\n";
	cout << "The average of grades is : " << fixed << setprecision(4) << Average << endl;
	



	return 0;
}
