the #43  :

it is about to change seconds to : days - hours - minutes - secondes




#include <iostream>
using namespace std;

int main()
{

	int SecondsPerDay = 24 * 60 * 60, SecondsPerHour = 60 * 60, SecondsPerMinute = 60, TotalSeconds;

	cout << "Pleas enter the totalseconds that you want to change : ";
	cin >> TotalSeconds;

	int NumberOfDays = TotalSeconds / SecondsPerDay;
	int remainder1 = TotalSeconds % SecondsPerDay;

	int NumberOfHours = remainder1 / SecondsPerHour;
	int remainder2 = remainder1 % SecondsPerHour;

	int NumberOfMinutes = remainder2 / SecondsPerMinute;
	int NumberOfSeconds = remainder2 % SecondsPerMinute;


	cout << NumberOfDays << ":" << NumberOfHours << ":" << NumberOfMinutes << ":" << NumberOfSeconds << endl;



	return 0;
}
