void Operation_Result(string& Operation, float& Number1, float& Number2)
{

	if (Operation == "+")
	{

		cout << "       The Result is : " << Number1 + Number2 << " \n\n";


	}
	else if (Operation == "-")
	{

		cout << "       The Result is : " << Number1 - Number2 << " \n\n";


	}
	else if (Operation == "*")
	{

		cout << "       The Result is : " << Number1 * Number2 << " \n\n";


	}
	else if (Operation == "/")
	{

		cout << "       The Result is : " << Number1 / Number2 << " \n\n";


	}
	else
	{

		cout << " The Operation is Wrong, Please try to use Just one of those Operations : \'+\' \'-\' \'*\' \'/\'" << endl << endl;

	}





}



int main()
{

	float Number1, Number2;
	string Operation;


	Operation_between_two_Num(Number1, Number2, Operation);
	Operation_Result(Operation, Number1, Number2);

	return 0;
}
