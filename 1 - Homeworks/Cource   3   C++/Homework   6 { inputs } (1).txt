P2: this programe do a + calculation




#include <iostream>
using namespace std;

int main()
{

	float Number1;
	float Number2;
	float Number3;

	cout << "Enter the first Number ?" << endl;
	cin >> Number1;

	cout << "\nEnter the second Number ? " << endl;
	cin >> Number2;

	cout << "\nEnter the therd Number ? " << endl;
	cin >> Number3;

	float sum = Number1 + Number2 + Number3;

	cout << endl << endl << endl << Number1 << "+\n";
	cout << Number2 << "+\n";
	cout << Number3 << "\n\n";
	cout << "---------------------------------------------- \n";
	cout << sum << endl << endl;

	return 0;
}
