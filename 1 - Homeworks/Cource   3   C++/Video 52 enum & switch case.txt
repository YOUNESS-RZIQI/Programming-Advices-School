#include <iostream>
using namespace std;

enum EnSystem_Color { Red = 1, Bleu = 2 , Yellow =3 , Green =4};

void System_Color_Menu()
{

	cout << "*****************************************************\n";
	cout << "*                    System Color                   *\n";
	cout << "*****************************************************\n";
	cout << "*                                                   *\n";
	cout << "*       1: Red                                      *\n";
	cout << "*       2: Bleu                                     *\n";
	cout << "*       3: Yellow                                   *\n";
	cout << "*       4: Green                                    *\n";
	cout << "*                                                   *\n";
	cout << "*****************************************************\n";
	cout << endl;
	cout << "           Choose one Color : ";


}

EnSystem_Color Asking_For_System_Color()
{

	int SC;
	cin >> SC;
	return (EnSystem_Color)SC;

}

int System_Result(EnSystem_Color v)
{

	switch (v)
	{
	case Red:
		return system("color 4F ");
		break;
	case Bleu:
		return system("color 1F");
		break;
	case Yellow:
		return system("color 6F");
		break;
	case Green:
		return system("color 2F");
		break;
	default:
		return system("color F0");
		break;
	}

}

int main()
{

	System_Color_Menu();

	cout << System_Result(Asking_For_System_Color());


	return 0;
}
