


#include <iostream>
#include <cmath>
using namespace std;

int main()
{

	int A = 10, B = 20;

	A++;   // Same as A = A+1 
	B--;   // Sane as B = B-1

	cout << "A = " << A << endl;
	cout << "B = " << B << endl;

	return  0;
}
