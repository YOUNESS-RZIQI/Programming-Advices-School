bool is_there_is_fraction(float Number)
{

	int Num1 = Number;
	Number = Number - Num1;

	if (Number > 0)
		return true;
	else
		return false;

}

bool is_leap_Year(const float & Year)
{

	if (!is_there_is_fraction((Year / 400)))
	{
		return true;
	}

	if (!is_there_is_fraction((Year / 4)) && is_there_is_fraction((Year / 100)))
	{
			return true;
	}

	return false;
}

int main()
{


	int Year = Input_Lib::Read_Positive_Number("Enter Number : ");

	if (is_leap_Year(Year))
	{
		cout << "\n\nYes leap Year\n\n";
	}
	else
	{
		cout << "\n\nNo Not a leap Year\n\n";
	}
