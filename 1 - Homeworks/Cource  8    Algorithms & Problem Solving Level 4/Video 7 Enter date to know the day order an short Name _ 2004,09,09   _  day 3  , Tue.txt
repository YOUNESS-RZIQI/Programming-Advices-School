short Operation(const short& Year, const short& Month, const short& Day)
{

	int a, y, m , day_order;

	a = (14 - Month) / 12;
	y = Year - a;
	m = Month + (12 * a) - 2;

	day_order = (Day + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;
	return day_order;
}

void DayNameOfWeek_Operatoin(const short& Year, const short& Month, const short& Day)
{

	short Day_Order = 0;
	string Days[]{ "Sun","Mun","Tue","Wedn","Thu","Fri","Sat" };

	Day_Order = Operation(Year, Month, Day);

	cout << "Day Order : " << Day_Order << "\n";
	cout << "Day Nam   : " << Days[Day_Order] << "\n";
						 
}

void DayNameOfWeek(const short& Year, const short& Month, const short& Day)
{
	cout << "--------------------------------------\n";
	cout << "\n\nDate      : " << Day << "/" << Month << "/" << Year << endl;
	DayNameOfWeek_Operatoin(Year, Month, Day);
	cout << "\n\n--------------------------------------\n";

}
