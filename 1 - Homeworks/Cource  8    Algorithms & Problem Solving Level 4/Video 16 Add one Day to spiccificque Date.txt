s_Date Increas_Date_by_one_Day(const int& Day_from_bi, const short& year)
{
	s_Date Date;

	short Remaining = (Day_from_bi)+1;
	int Month_Days = 0;

	Date.Month = 1;
	Date.Year = year;


	while (true)
	{
		Month_Days = NumberOfDaysInAMonth(Date.Year, Date.Month);

		if (Remaining >= Month_Days)
		{
			Remaining -= Month_Days;

			Date.Month++;

			if (Date.Month > 12)
			{
				Date.Month = 1;
				Date.Year++;
			}
		}
		else
		{
			Date.Day = Remaining;
			break;
		}

	}


	return Date;
}
