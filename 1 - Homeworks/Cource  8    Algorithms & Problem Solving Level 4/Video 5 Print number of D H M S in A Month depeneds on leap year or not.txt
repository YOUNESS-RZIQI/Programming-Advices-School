int NumberOfDaysInAMonth(short Year , short Month )
{
	if(Month <= 0 || Month > 12)
		return 0;

	if (Month == 2)
		return Check_Lib::is_leap_Year(Year) ? 29 : 28;

	/*else if (Month == 1 || Month == 3 || Month == 5 || Month == 7 || Month == 8 || Month == 10 || Month == 12)
		return 31;*/

	
	short arr31Days[] = { 1,3,5,7,8,10,12 };

	for (short i = 0; i < 7; i++)
	{
		if (arr31Days[i] == Month)
			return 31;
	}

	return 30;	
}

int NumberOfHourssInAMonth(short Year, short Month)
{
	return NumberOfDaysInAMonth(Year, Month) * 24;
}

int NumberOfMinutesInAMonth(short Year, short Month)
{
	return NumberOfHourssInAMonth(Year, Month) * 60;
}

int NumberOfSecondsInAMonth(short Year, short Month)
{
	return NumberOfMinutesInAMonth(Year , Month) * 60 ;
}

void Print_if_Leap_or_Nor_Leap_Year_D_H_M_S(short Year, short Month)
{
	cout << "Number of Days     in Month [" << Month << "] is : " << NumberOfDaysInAMonth(Year, Month)    << endl;
	cout << "Number of Hours    in Month [" << Month << "] is : " << NumberOfHourssInAMonth(Year , Month)  << endl;
	cout << "Number of Minutes  in Month [" << Month << "] is : " << NumberOfMinutesInAMonth(Year, Month) << endl;
	cout << "Number of Seconds  in Month [" << Month << "] is : " << NumberOfSecondsInAMonth(Year, Month) << endl;

}

int main()
{
	Screen_Lib::Red_Screen();

	short Year = Input_Lib::Read_Positive_Number("Please enter the year : ");
	short Month = Input_Lib::Read_Positive_Number("Please enter the Month : ");

	Print_if_Leap_or_Nor_Leap_Year_D_H_M_S(Year , Month);


	system("pause>0");
	return 0;
}
