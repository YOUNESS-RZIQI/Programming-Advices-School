#include <iostream>
#include <cstdlib>
using namespace std;



int Randam_Number_In_Range(int From, int To)
{

	int RandNum = rand() % (To - From + 1) + From;

	return RandNum;
}




int main()
{

	srand((unsigned)time(NULL));

	cout << Randam_Number_In_Range(12, 14) << endl ;
	cout << Randam_Number_In_Range(12, 14) << endl ;
	cout << Randam_Number_In_Range(12, 14) << endl ;


	return 0;
}
