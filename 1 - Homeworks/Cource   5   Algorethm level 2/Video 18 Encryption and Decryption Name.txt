string Encryption_Name(string &Name, int Encryption_Key)
{

	int Digits_Number = Name.length();

	for (int i = 0; i < Digits_Number; i++)
	{

		Name[i] += Encryption_Key;

	}

	return  Name ;

}
								   
string Decryption_Name(string &Name, int Encryption_Key)
{

	int Digits_Number = Name.length();

	for (int i = 0; i < Digits_Number; i++)
	{

		Name[i] -= Encryption_Key;

	}

	return  Name;

}

#include <iostream>
#include <string>
using namespace std;


string Enter_Your_Name()
{

	string Text = "";

	cout << "Please Enter Your Name : ";
	getline(cin, Text);

	return Text;

}

string Encryption_Name(string Text, short Encryption_Key)
{

	int Digits_Number = Text.length();

	for (int i = 0; i < Digits_Number; i++)
	{

		Text[i] = char( (int)Text[i] + Encryption_Key);

	}

	return  Text;

}
								   	
string Decryption_Name(string Text, short Encryption_Key)
{


	int Digits_Number = Text.length();

	for (int i = 0; i < Digits_Number; i++)
	{

		Text[i] = char((int)Text[i] - Encryption_Key);

	}

	return  Text;

}
								   

int main()
{

	short Encryption_Key = 2;
	string Text = Enter_Your_Name();

	string Text_After_Encryption = Encryption_Name(Text, Encryption_Key);
	string Text_After_Decryption = Decryption_Name(Text_After_Encryption, Encryption_Key);
