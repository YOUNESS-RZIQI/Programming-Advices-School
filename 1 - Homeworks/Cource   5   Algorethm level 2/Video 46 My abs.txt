int Enter_Positive_Number()
{

	int Number = 0;

	do
	{

		cout << "Please enter the Number you Want to add to Array : ";
		cin >> Number;
		cout << endl;


	} while (Number <= 0);

}

int Randam_Number_In_Range(int From, int To)
{
	int RandNum = rand() % (To - From + 1) + From;
	return RandNum;
}

void Print_Array(int Arr_Source[100], short Arr_Source_Length)
{


	for (int i = 0; i < Arr_Source_Length; i++)
	{

		cout << Arr_Source[i] << "  ";

	}

	cout << endl;


}



void Fill_Array_with_N_Randum_Number(int Arr_Source[100], short& Arr_Source_Length)
{

	Arr_Source_Length = Enter_Positive_Number();


	for (int i = 0; i < Arr_Source_Length; i++)
	{

		Arr_Source[i] = Randam_Number_In_Range(-100, 100);

void Add_Array_Element(int UserNumber, int Arr_Source[100], int& Arr_Source_Length)
{

	Arr_Source_Length++;

	Arr_Source[Arr_Source_Length - 1] = UserNumber;

}

void Copye_if_Odd_Number(int Arr_Source[100], int Arr_Copye[100], int Arr_Source_Length, int& Arr_Coype_Length)
{

	for (int i = 0; i < Arr_Source_Length; i++)
	{
		if (Arr_Source[i] % 2 != 0)
		{
			Add_Array_Element(Arr_Source[i], Arr_Copye, Arr_Coype_Length);
		}

	}

}

bool Check_if_odd_Number(int Number)
{

	if (Number % 2 != 0)
		return true;
	else
		false;
}

float  My_ABS(float Number)
{

	if (Number < 0)
		return Number * -1;
	else
		return Number;
	
}
