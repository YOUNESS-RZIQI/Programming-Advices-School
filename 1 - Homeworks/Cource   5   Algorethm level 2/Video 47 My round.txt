float Number_Enter()
{
	float Num = 0;
	
	cout << "Please enter The Number : ";
	cin >> Num;

	return Num;
}

float Get_Fraction(float Number)
{
	return Number - int(Number);
}

float My_Round(float Number)
{

	int IntNum = int(Number);

	float Fraction = Get_Fraction(Number);

	if (abs(Fraction) >= .5)
	{
		if (Number > 0)
			return ++IntNum;
		else
			return --IntNum;
	}
	else
		return IntNum;
			

}
