#include <iostream>
#include <cstdlib>
using namespace std;



int Enter_Positive_Number()
{

	int Number = 0;

	do
	{

		cout << "Please enter the Number you Want to add to Array : ";
		cin >> Number;
		cout << endl;


	} while (Number <= 0);


	return Number;
}

int Randam_Number_In_Range(int From, int To)
{
	int RandNum = rand() % (To - From + 1) + From;
	return RandNum;
}

void Print_Array(int Arr[100], int ArrLength)
{


	for (int i = 0; i < ArrLength; i++)
	{

		cout << Arr[i] << " ";

	}

	cout << endl;


}








void Add_Array_Element(int UserNumber, int Arr[100], short& ArrLength)
{


	ArrLength++;

	Arr[ArrLength - 1] = UserNumber;


}

void Copye_Arr_Using_Add_Array_Element( int Arr[100], int Arr2[100], short ArrLength , short &Arr2Length )
{

	for ( int i = 0 ; i < ArrLength; i++)
	{

		Add_Array_Element( Arr[i], Arr2, Arr2Length);

	}

}

	ArrLength = Enter_Positive_Number();

	}

}



int main()
