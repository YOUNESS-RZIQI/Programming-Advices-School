#include <iostream>
using namespace std;


float Enter_N()
{

	float N;

	cout << "      Please Enter N That will go to 1 : ";
	cin >> N;

	return N;
}


void Print_From_1_To_N_For(float N)
{

	cout << "\n\n   For :\n";

	for (int i = 1; N >= i; i++)
	{

		cout << i << endl;

	}

}

void Print_From_1_To_N_While(float N)
{

	cout << "\n\n   While :\n";

	int i = 1;

	while (N >= i)
	{
		
		cout  << i << endl;
		i++;

	}

}

void Print_From_1_To_N_Do_While(float N)
{

	cout << "\n\n   Do While :\n";

	int i = 1;

	do
	{
		
		cout  << i << endl;
		i++;

	} while (N >= i);

}

int main()
{

	int N = Enter_N();
	Print_From_1_To_N_For(N);
	Print_From_1_To_N_While(N);
	Print_From_1_To_N_Do_While(N);

	return 0;
}
