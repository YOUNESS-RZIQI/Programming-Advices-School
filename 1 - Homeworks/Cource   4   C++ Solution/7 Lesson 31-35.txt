#include <iostream>
using namespace std;


struct stCoin
{
	float Penny, Nickel, Dime, Quarter, Dollar;
};

void Enter_Coins(stCoin &Coin)
{

	cout << "Please Enter Pennys  : ";
	cin >> Coin.Penny;

	cout << "Please Enter Nickel  : ";
	cin >> Coin.Nickel;

	cout << "Please Enter Dime    : ";
	cin >> Coin.Dime;

	cout << "Please Enter Quarter : ";
	cin >> Coin.Quarter;

	cout << "Please Enter Dollar  : ";
	cin >> Coin.Dollar;

}

float Change_All_Coins_To_Penny(stCoin Coin)
{

	float Change = 0;

	Change = (Coin.Penny * 1) + (Coin.Nickel * 5) + (Coin.Dime * 10) + (Coin.Quarter * 25) + (Coin.Dollar * 100);

	return Change;

}

float Change_All_From_Penny_To_Dollar(stCoin Coin)
{

	float Change = 0;

	Change = Change_All_Coins_To_Penny(Coin) / 100;

	return Change;

}



int main()
{

	stCoin Coin;

	Enter_Coins(Coin);

	cout << "\n\nTotal Pennys  : " << Change_All_Coins_To_Penny(Coin) << endl ;
	cout << "Total Dollars : " << Change_All_From_Penny_To_Dollar(Coin) << endl << "\n\n";



	return 0;
}
