
#include <iostream>
#include <iomanip>
#include <vector>
using namespace std;


int main()
{

	int Arr[4] = { 1,2,3,4 };
	int * ptr;
	ptr = Arr;

	cout << "The addressees : \n\n";

	for (int i = 0 ; i < 4 ; i++)
	{

		cout << ptr + i << endl;

	}


	cout << "\n\nThe Values : \n\n";

	for (int i = 0; i < 4; i++)
	{

		cout << *(ptr + i) << endl;

	}


	return 0;
}
