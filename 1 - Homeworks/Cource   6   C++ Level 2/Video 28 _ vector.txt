


// Vector



#include <iostream>
#include <iomanip>
#include <vector>
using namespace std;




int main()
{

	vector <int> V_Numbers = { 10 , 20 , 30 , 40 , 50 };

	cout << "\n\nThe Numbers in the vector are : ";

	for (int &Number : V_Numbers)
	{
		cout << Number << " ";
	}


	cout << endl;

	return 0;
}
