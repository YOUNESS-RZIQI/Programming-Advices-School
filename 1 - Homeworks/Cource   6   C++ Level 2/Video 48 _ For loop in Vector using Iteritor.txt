
#include <iostream>
#include <iomanip>
#include <vector>
using namespace std;


int main()
{

	vector <int> Numbers = { 1,2,3 };

	vector <int>::iterator  i;

	for (i = Numbers.begin(); i != Numbers.end(); i++)
	{
		cout << *i << " ";
	}

	cout << "\n\n\n";

	return 0;
}
