int Randam_Number_In_Range(int From, int To)
{

	int RandNum = rand() % (To - From + 1) + From;

	return RandNum;
}

void Fill_Two_Dimensional_Arrays(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			Arr[i][j] = Randam_Number_In_Range(1, 100);
		}

	}


}

int Sum_Coloms_of_Matrix(int Arr[3][3], const short& Row_Number, const short& Coloms)
{

	int sum = 0;
	for (int j = 0; j < Row_Number; j++)
	{
		sum += Arr[j][Coloms];
	}

	return sum;
}

void Colms_sum_to_Array(int Arr[3][3], const short& Rows, const short& Coloms , int arr[3])
{

	for (int i = 0; i < Coloms; i++)
	{
		arr[i] = Sum_Coloms_of_Matrix(Arr, Rows , i);
	}

}

void Print_Matrix(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			//printf("%d\t", Arr[i][j]);
			cout << setw(3) << Arr[i][j] << "    ";
		}

		cout << endl;

	}


}

void Print_Array(int Arr[3], short length)
{

	for (int i = 0; i < length; i++)
