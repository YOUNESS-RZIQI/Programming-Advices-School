char Invert_Letter(char  Letter)
{

	return isupper(Letter) ? tolower(Letter) : toupper(Letter);

}

char Read_char(string Message)
{

	char Word = 'a';
	cout << Message;
	cin >> Word;

	return Word;
}

short Count_Character(const string& Word, const char& Character, bool  Matche = true)
{

	short Count = 0;

	for (short i = 0; i < Word.length(); i++)
	{
		if (Matche)
		{
			if (Word[i] == Character)
				Count++;
		}
		else if (toupper(Word[i]) == toupper(Character))
			Count++;

	}

	return Count;
}
