#include <iostream>
#include <iomanip>
#include <vector>
#include <string>
#include <fstream>
#include "Input_Lib.h"
using namespace std;

vector <string> Split_String( string & s, const string & delim)
{

	vector <string> vNames;

	short pos = 0;
	string Word;

	while ((pos = s.find(delim)) != std::string::npos)
	{

		Word = s.substr(0, pos);

		if (Word != "")
		{
			vNames.push_back(Word);
		}

		s.erase(0, pos + delim.length());

	}

	if (s != "")
	{
		vNames.push_back(s);
	}


	return vNames;
}

string Join_String(vector <string>& vNames, const string& delim)
{

	string S1 = "";

	for (string& s : vNames)
	{
		S1 = S1 + s + delim;
	}

	return S1.substr(0, S1.length() - delim.length());
}

string Replace_word( string & s ,const string & Replace_this, const string & Replace_with )
{

	vector <string> vNames;

	vNames = Split_String(s, " ");

	for (short i = 0; i < vNames.size(); i++)
	{

		if (vNames[i] == Replace_this)
			vNames[i] = Replace_with;

	}
	
	string s2 = Join_String(vNames, " ");

	return s2;
}


int main()
{


	string s = Input_Lib::Ask_for_string("Enter The string Please : \n");

	string Replace_this = Input_Lib::Ask_for_string("Enter The Word you want to change Please : \n");
	string Replace_with = Input_Lib::Ask_for_string("with what ? \n");

	cout << "\n\n" << Replace_word(s, Replace_this, Replace_with);

	cout << "\n\n";


	system("pause>0");
}

string Replace_word_using_build_in_Function( string & s ,const string & Replace_this, const string & Replace_it_with)
{

	short pos = s.find(Replace_this);

	while(pos != std::string::npos)
	{

		s.replace(pos, Replace_this.length(), Replace_it_with);
		pos = s.find(Replace_this);

	}

	return s;
}
