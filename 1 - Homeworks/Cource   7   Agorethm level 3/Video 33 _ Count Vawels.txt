bool is_Vawel(char  Ch1)
{

	Ch1 = tolower(Ch1);

	return ((Ch1 == 'a') || (Ch1 == 'e') || (Ch1 == 'i') || (Ch1 == 'o') || (Ch1 == 'u'));

}

short Count_Vawels(string S)
{

	short Counter = 0;

	for (int i = 0; i < S.length(); i++)
	{

		if (is_Vawel(S[i]))
			Counter++;

	}

	return Counter;
}

int main()
{

	cout << "Please enter your String : ";

	cout << "\n\nThere is " << Count_Vawels(s) << " in your String\n\n";

	


	return 0;
}
