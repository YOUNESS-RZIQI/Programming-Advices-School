string Change_First_leter_to_Capital( string  Word)
{

	bool is_First_letter = true;

	for (int i = 0; i < Word.length(); i++)
	{

		if (Word[i] != ' ' && is_First_letter)
		{
			Word[i] = toupper(Word[i]);
		}

		is_First_letter = (Word[i] == ' ' ? true : false);

	}

	return Word;
}

	string Word = Read_String();

	Word = Change_First_leter_to_Capital(Word);

	cout << "\n\nThe Result is : \n";
	cout << Word << "\n\n";


	return 0;
}
