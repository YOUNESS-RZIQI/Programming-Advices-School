int Randam_Number_In_Range(int From, int To)
{

	int RandNum = rand() % (To - From + 1) + From;

	return RandNum;
}

void Fill_Matrix_W_Randum_Number(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			Arr[i][j] = Randam_Number_In_Range(1, 10);
		}

	}


}

void Print_Matrix(int Arr[3][3], const short& Rows, const short& Coloms)
{

	for (int i = 0; i < Rows; i++)
	{

		for (int j = 0; j < Coloms; j++)
		{
			printf("%0*d\t", 2, Arr[i][j]);
			//cout << setw(3) << Arr[i][j] << "    ";
		}

		cout << endl;

	}


}


bool is_Number_exest_in_Matrix(int Matrix[3][3], const short& Rows, const short& Colms, const short& Number_to_Check)
{

	for (int i = 0; i < Rows; i++)
	{
		for (int j = 0; j < Colms; j++)
		{
			if (Matrix[i][j] == Number_to_Check)
				return true;
		}
	}

	return false;
}

{

}
