
#include <iostream>

using namespace std;

enum en_What_to_Count {Capital , Small , All};

short Count_Letters(const string& Word, en_What_to_Count What_to_Count = en_What_to_Count::All)
{

	if (What_to_Count == en_What_to_Count::All)
		return Word.length();

	short Counter = 0;

	for (short i = 0; i < Word.length(); i++)
	{
		if (What_to_Count == en_What_to_Count::Capital)
		{
			if (isupper(Word[i]))
				Counter++;
		}
		else if (What_to_Count == en_What_to_Count::Small)
		{
			if (islower(Word[i]))
				Counter++;
		}
		
	}

	return Counter;
}

short Count_Capital_Letters(const string& Word)
{
	short Counter = 0;

	for (short i = 0; i < Word.length(); i++)
	{
		if (isupper(Word[i]))
			Counter++;
	}

	return Counter;
}

short Count_Small_Letters(const string& Word)
{
	short Counter = 0;

	for (short i = 0; i < Word.length(); i++)
	{
		if (islower(Word[i]))
			Counter++;
	}

	return Counter;
}

	return 0;
}
