string Join_String( vector <string> & vNames, const string& delim)
{

	string S1 = "";

	for (string & s : vNames)
	{
		S1 = S1 + s + delim ;
	}

	return S1.substr(0,S1.length() - delim.length());
}

int main()
{

	vector <string> vNames = { "Ali","Maher","Fadi","Saqre" };

	

	cout << "Vector after Join : \n";
	cout << Join_String (vNames,"###")  << endl;
